USE CENTROS_EDUCTAIVOS

CREATE TABLE ESPECIALIDAD (
CODESP INT,
NOMBRE VARCHAR (50) NOT NULL,
LAGA VARCHAR (50) NOT NULL,
PRIMARY KEY (CODESP)
) ENGINE InnoDB;

CREATE TABLE CICLOFP (
CODCICLO varchar(3),
CODESP INT,
NOMBRE VARCHAR (10) NOT NULL,
NIVEL VARCHAR (8) NOT NULL,
HORAS TINYINT,
PRIMARY KEY (CODCICLO),
FOREIGN KEY (CODESP) REFERENCES ESPECIALIDAD (CODESP)
) ENGINE InnoDB;


CREATE TABLE CENTROSIES (
CODCENTRO varchar(3),
NOMBRE VARCHAR (10) NOT NULL,
UBICACION VARCHAR (20) NOT NULL,
DIRECCION VARCHAR(50),
PRIMARY KEY (CODCENTRO)
) ENGINE InnoDB;

CREATE TABLE ALUMNOS (
N_EXP INT,
CODCENTRO varchar(3),
NOMBRE VARCHAR (10) NOT NULL,
APELLIDO VARCHAR (10) NOT NULL,
PRIMARY KEY (N_EXP),
FOREIGN KEY (CODCENTRO) REFERENCES CENTROSIES (CODCENTRO)
) ENGINE InnoDB;

CREATE TABLE ALUMNOS_CICLOS (
N_EXP varchar(3),
CODCICLO VARCHAR (3) NOT NULL,
NOTA_FINAL TINYINT,
PRIMARY KEY (N_EXP, CODCICLO),
FOREIGN KEY (N_EXP) REFERENCES ALUMNOS (N_EXP),
FOREIGN KEY (CODCICLO) REFERENCES CICLOFP (CODCICLO)
) ENGINE InnoDB;


CREATE TABLE CENTRO_P (
CODCENTRO varchar(3),
DOTACION INT,
ANO VARCHAR (4),
PRIMARY KEY (CODCENTRO),
FOREIGN KEY (CODCENTRO) REFERENCES CENTROSIES (CODCENTRO)
) ENGINE InnoDB;



CREATE TABLE CENTRO_C (
CODCENTRO varchar(3),
AYUNTAMIENTO VARCHAR (15),
PRIMARY KEY (CODCENTRO),
FOREIGN KEY (CODCENTRO) REFERENCES CENTROSIES (CODCENTRO)
) ENGINE InnoDB;
